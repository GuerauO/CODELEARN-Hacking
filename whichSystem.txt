import re, sys, subprocess
def get_ttl(ip): 
	proc = subprocess.Popen(["/usr/bin/ping -c1 %s" % ip, ""], stdout=subprocess.PIPE, shell=True)
	(out, err ) = proc.communicate() 
	print(out)
	out = out.split() 
	out = out[12].decode('utf-8') 
	print(out)
	ttl_value = re.findall(r"\d{1,3}", out) 
	print(ttl_value) 
	return ttl_value[0]

def get_os(ttl):
	ttl = int(ttl) 
	if ttl >= 0 and ttl <= 64: 
		return "Linux" 
	elif ttl >= 65 and ttl <= 128: 
		return "Windows" 
	else:
		return "Not found" 

if len(sys.argv) != 2: 
	print("\n[!] Use: python3 " + sys.argv[0] + "\n") 
	sys.exit(1) # salida con cÃ³digo de estado no existoso
if __name__ == "__main__": 
	ip_address = sys.argv[1] 
	ttl = get_ttl(ip_address) 
	os_name = get_os(ttl) 
	print("%s (ttl â†’ %s): %s" % (ip_address, ttl, os_name))
